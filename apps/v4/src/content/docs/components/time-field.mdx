## Usage

```tsx
import {
  TimeField,
  TimeFieldAmPm,
  TimeFieldHours,
  TimeFieldMinutes,
  TimeFieldSeconds,
  TimeFieldSeparator,
} from "@/components/ui/time-field";
```

```tsx
<TimeField>
  <TimeFieldHours />
  <TimeFieldSeparator />
  <TimeFieldMinutes />
  <TimeFieldSeparator />
  <TimeFieldSeconds />
  <TimeFieldAmPm />
</TimeField>
```
