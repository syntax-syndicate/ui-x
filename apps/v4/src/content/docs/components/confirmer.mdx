## Usage

```tsx
import { confirm } from "@/components/ui/confirmer";
```

```tsx
confirm({
  title: "Are you absolutely sure?",
  description:
    "This action cannot be undone. This will permanently delete your account and remove your data from our servers.",
}).then((confirmed) => {
  // ...
});
```
