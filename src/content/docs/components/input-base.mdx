## Usage

```tsx
import {
  InputBase,
  InputBaseAdornment,
  InputBaseControl,
  InputBaseInput,
} from "@/components/ui/input-base"
```

```tsx
<InputBase>
  <InputBaseAdornment>@</InputBaseAdornment>
  <InputBaseControl>
    <InputBaseInput />
  </InputBaseControl>
</InputBase>
```
