## Usage

```tsx
import {
  PasswordInput,
  PasswordInputAdornmentToggle,
  PasswordInputInput,
} from "@/components/ui/password-input"
```

```tsx
<PasswordInput>
  <PasswordInputInput placeholder="Password" />
  <PasswordInputAdornmentToggle />
</PasswordInput>
```
